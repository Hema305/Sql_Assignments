1)	WRITE A SQL STATEMENT TO DISPLAY THE LOWEST PAID EMPLOYEE'S (NAME , SALARY , DEPARTMENT NAME)

SELECT E.FIRST_NAME,E.SALARY,D.DEPARTMENT_NAME
    -> FROM EMPLOYEES E JOIN DEPARTMENTS D
    -> ON E.DEPARTMENT_ID=D.DEPARTMENT_ID AND SALARY=(SELECT MIN(SALARY) FROM EMPLOYEES);

2)	LIST MINIMUM SALARY FOR EACH DEPARTMENT

SELECT DEPARTMENT_ID,MIN(SALARY)
    -> FROM EMPLOYEES
    -> GROUP BY DEPARTMENT_ID;

3)	WRITE A QUERY  BASED ON  FOLLOWING RESULT. 

SELECT E.EMPLOYEE_ID, E.FIRST_NAME,E.JOB_ID,E.SALARY,D.DEPARTMENT_ID,D.DEPARTMENT_NAME
    -> FROM EMPLOYEES E JOIN DEPARTMENTS D
    -> ON E.DEPARTMENT_ID=D.DEPARTMENT_ID;

4)	LIST ALL THE EMPLOYEES WHO ARE WORKING IN FORD’S(FINANCE) DEPARTMENT.
SELECT E.EMPLOYEE_ID, E.FIRST_NAME,E.JOB_ID,E.SALARY,E.HIRE_DATE,E.MANAGER_ID,D.DEPARTMENT_ID,D.DEPARTMENT_NAME
    -> FROM EMPLOYEES E JOIN DEPARTMENTS D
    -> ON E.DEPARTMENT_ID=D.DEPARTMENT_ID AND DEPARTMENT_NAME='Finance';
